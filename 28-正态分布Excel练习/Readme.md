```
import numpy as np
from math import factorial

############################################################
# 构建函数模拟二项分布
############################################################
def binomial_distribution(n, p):
    """
    :param n: 实验次数，是一个正整数
    :param p: 每一次实验成功的概率，取值为0-1之间的小数
    :return: 存储实验成功次数对应的概率的列表（成功次数为列表的各个索引）
    """
    list_x = [(factorial(n) / factorial(i) / factorial(n-i)) * np.power(p, i) * np.power(1-p, n-i) for i in
              range(n+1)]
    return list_x

############################################################
# 定义标准正态分布的概率密度函数（pdf）
############################################################
# 也可直接调用用 scipy.stats.norm.pdf，和以下函数定义的内容相同
# mean = 0
# std = 1
def norm_distribute(x, mean, std):
    """
    :param x: 随机变量取值，是一个标量数值
    :return: 概率密度函数对应的取值，是一个标量数值
    """
    return (1 / (np.power(2 * np.pi, 0.5) * std)) * np.power(np.e, -(np.power(x - mean, 2)) / (2 * np.power(std, 2)))

############################################################
# 对比二项分布和正态分布
############################################################
# 二项分布 ==================================================
n = 10
p = 0.5
mean = n * p
std = (n * p * (1-p)) ** 0.5
list_01 = binomial_distribution(n=n, p=p)
p_01_5 = list_01[5]
print("二项分布随机变量取值为5的概率是：", p_01_5)

# 正态分布 ==================================================
# 根据上述二项分布的均值和标准差定义正态分布的概率密度函数
p_nd_5 = norm_distribute(x=5, mean=mean, std=std)
print("正态分布随机变量取值为5时的PDF取值是：", p_nd_5)

# 二项分布随机变量取值为5的概率是： 0.24609375
# 正态分布随机变量取值为5时的PDF取值是： 0.252313252202016
# 可有看出二者的取值比较接近
```
